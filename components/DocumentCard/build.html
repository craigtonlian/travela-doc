                    children: [

                                      maxLines: 1,

                                        )
                                      ];
                                    },
                                    onSelected: (value) async {
                                      switch (value) {
                                        case DocumentCardEnum.open:
                                          bool result =
                                              await DocumentController.open(

                                                      .where(documentId));

                                          if (!result) {
                                            await Confirmation.confirmDialog(
                                                'Cannot Download',
                                                'The file cannot be downloaded right now. Try again later!',

                                        case DocumentCardEnum.edit:

                                                controller: controller,

                                          await Confirmation
                                              .confirmCancelDialog(
                                                  'Confirm Delete?',
                                                  'Warning: Deleted data is permanently lost and cannot be '
                                                      'recovered. Proceed?',
                                                  () async {
                                            await controller
                                                .deleteDocument(documentId)
                                                .then((value) => Get.back());

                                    },
