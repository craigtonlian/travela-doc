Parses the Weather object and returns the WeatherType classification, which tells the WeatherBg object what animation to render.

static WeatherType? dispatchWeather(Weather? weather) {
  if (weather == null) {
    return null;
  }

  int code = weather.weatherConditionCode ?? -1;

  if (code == -1) {
    return null;
  }

  if (200 <= code && code <= 232) {
    return WeatherType.thunder;
  } else if (300 <= code && code <= 321) {
    return WeatherType.lightRainy;
  } else if (500 <= code && code <= 531) {
    return WeatherType.heavyRainy;
  } else if (600 <= code && code <= 622) {
    return WeatherType.middleSnow;
  } else if (code == 741 || code == 701) {
    return WeatherType.foggy;
  } else if (code == 711 || code == 721) {
    return WeatherType.hazy;
  } else if (701 <= code && code <= 781) {
    // anything else defaults to dusty
    return WeatherType.dusty;
  } else if (code == 800) {
    DateTime? timeOfObservation = weather.date;

    if (timeOfObservation == null || timeOfObservation.hour < 18) {
      return WeatherType.sunny;
    } else {
      return WeatherType.sunnyNight;
    }
  } else if (801 <= code && code <= 804) {
    DateTime? timeOfObservation = weather.date;

    if (timeOfObservation == null || timeOfObservation.hour < 18) {
      return WeatherType.cloudy;
    } else {
      return WeatherType.cloudyNight;
    }
  }

  return null;
}
