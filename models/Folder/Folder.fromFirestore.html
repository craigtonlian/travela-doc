factory Folder.fromFirestore(DocumentSnapshot<Map<String, dynamic>> snapshot,
    SnapshotOptions? options) {
  final data = snapshot.data();

  return Folder._(
      folderId: snapshot.id,
      userId: UserController.to.obsTravelaUser().userId,
      name: data?['name'],
      isStarred: data?['isStarred'],
      color: Color.fromARGB(data?['color'][3], data?['color'][0],
          data?['color'][1], data?['color'][2]),
      storageUrl: data?['storageUrl'],
      usersList: List.from(data?['usersList']),
      foldersList: List.from(data?['foldersList']),
      documentsList: List.from(data?['documentsList']),
      creationDate: data?['creationDate']?.toDate(),
      lastUpdated: data?['lastUpdated']?.toDate());
}
