factory PlaceOfInterest.fromJSON(Map<String, dynamic> map) {
  return PlaceOfInterest._(
      map['name'],
      [
        map['geometry']['location']['lat'],
        map['geometry']['location']['lng']
      ],
      map['place_id'],
      double.parse(map['rating'].toString()),
      map['business_status']);
}
