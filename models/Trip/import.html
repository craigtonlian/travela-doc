static Trip import(Map<String, dynamic> map) {
  return Trip._(
      tripId: map['tripId'] ?? (throw ArgumentError('tripId is required')),
      userId: map['userId'] ?? (throw ArgumentError('userId is required')),
      location: Location.fromFirestore(
          map['location'] ?? (throw ArgumentError('location is required'))),
      isStarred:
          map['isStarred'] ?? (throw ArgumentError('isStarred is required')),
      startDate: map['startDate']?.toDate() ??
          (throw ArgumentError('startDate is required')),
      endDate: map['endDate']?.toDate() ??
          (throw ArgumentError('endDate is required')),
      usersList: List.from(map['usersList']),
      foldersList: List.from(map['foldersList']),
      documentsList: List.from(map['documentsList']),
      itinerariesList: List.from(map['itinerariesList']),
      expensesList: List.from(map['expensesList']),
      creationDate: map['creationDate']?.toDate() ??
          (throw ArgumentError('creationDate is required')),
      lastUpdated: map['lastUpdated']?.toDate() ??
          (throw ArgumentError('lastUpdated is required')));
}
