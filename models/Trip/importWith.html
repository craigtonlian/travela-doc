Parses a JSON response from the Server, and allow modifications to the attributes of the JSON repsonse.

Parameters:

  • map - JSON response object
  • tripId - The Trip ID of this Trip
  • userId - The User ID of the creator of the Trip
  • location - Location of the Trip
  • isStarred - The Starred status of the Trip
  • startDate - The start date of the Trip
  • endDate - The end date of the Trip
  • usersList - The list of Users who have access to this Trip
  • foldersList - The list of Folders nested within this Trip
  • documentsList - The list of Documents nested within this Trip
  • itinerariesList - The list of Itineraries nested within this Trip
  • expensesList - The list of Expenses nested within this Trip
  • lastUpdated - A running tally of the last time the Trip was updated

static Trip importWith(Map<String, dynamic> map,
    {String? tripId,
    String? userId,
    Location? location,
    bool? isStarred = false,
    DateTime? startDate,
    DateTime? endDate,
    List<String>? usersList,
    List<String>? foldersList,
    List<String>? documentsList,
    List<String>? itinerariesList,
    List<String>? expensesList,
    DateTime? lastUpdated}) {
  return Trip._(
      tripId: map['tripId'] ?? tripId,
      userId: map['userId'] ?? userId,
      location: Location.fromJSON(
          map['location'] ?? Location.fromJSON(map['location'])),
      isStarred: map['isStarred'] ?? isStarred,
      startDate: map['startDate']?.toDate() ?? startDate,
      endDate: map['endDate']?.toDate() ?? endDate,
      usersList: List.from(map['usersList']),
      foldersList: List.from(map['foldersList']),
      documentsList: List.from(map['documentsList']),
      itinerariesList: List.from(map['itinerariesList']),
      expensesList: List.from(map['expensesList']),
      lastUpdated: map['lastUpdated']?.toDate() ?? lastUpdated,
      creationDate: map['creationDate']?.toDate() ??
          (throw ArgumentError('No valid creation date')));
}
