Handles the exceptions thrown by authenticating methods above, and returns a popup with a short message on why the authentication method had failed.

e represents the error to handle, while context represents the enclosing context to show the popup on.

static void handle(FirebaseAuthException e, BuildContext context) {
  String err;

  switch (e.code) {
    case 'invalid-email':
      err = 'An invalid email was used. Try again.';
      break;
    case 'user-disabled':
      err = 'Account tied to email was suspended.';
      break;
    case 'user-not-found':
      err =
          'No account is associated with email. Sign up first before logging in.';
      break;
    case 'wrong-password':
      err = 'Wrong Password. Try again.';
      break;
    case 'email-already-in-use':
    case 'account-exists-with-different-credential':
      err =
          'Email is tied to another account. Try logging in with the other account instead.';
      break;
    case 'invalid-credentials':
      err = 'Wrong credentials are provided. Try again.';
      break;
    case 'operation-not-allowed':
      err = 'Contact the administrator for access to action.';
      break;
    case 'weak-password':
      err = 'Your password is too weak. Try again.';
      break;
    case 'network-request-failed':
      err =
          'Check your Internet connection to see if you are online. If so, there might be a '
          'server error. Try again in a while!';
      break;
    case 'ERROR_MISSING_GOOGLE_AUTH_TOKEN':
    default:
      err =
          'Unknown error has occurred. Contact the administrator for more information.\n\n'
          'Error code: ${e.code}';
      break;
  }

  // alert the user if there are any errors
  Confirmation.confirmDialog('Error!', err, () => Get.back());
}
