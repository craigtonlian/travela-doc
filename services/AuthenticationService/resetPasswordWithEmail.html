Allows the user to reset their email-based account passwords through a password reset email.

Requires the enclosing BuildContext context to properly show the popup. It also requires the email to reset password for.

Implementation

static Future<void> resetPasswordWithEmail(
    BuildContext context, String email) async {
  var isNormal = true.obs;

  await FirebaseAuth.instance
      .sendPasswordResetEmail(email: email)
      .onError((FirebaseAuthException error, stackTrace) {
    isNormal.value = false;
    AuthenticationService.handle(error, context);
  });

  if (isNormal.value) {
    Confirmation.confirmDialog(
        'Reset Password', 'Check your email for the email reset link!', () {
      Get.offAll(() => SplashScreen());

      showModalBottomSheet(
          clipBehavior: Clip.hardEdge,
          backgroundColor: Theme.of(context).colorScheme.background,
          constraints: BoxConstraints.expand(
              height: MediaQuery.of(context).size.height),
          useSafeArea: true,
          isScrollControlled: true,
          elevation: 3,
          shape: RoundedRectangleBorder(
            borderRadius: BorderRadius.only(
              topLeft: Radius.circular(28),
              topRight: Radius.circular(28),
            ),
          ),
          context: context,
          builder: (BuildContext context) => SplashOverlay(),
          showDragHandle: true);
    });
  }
}
