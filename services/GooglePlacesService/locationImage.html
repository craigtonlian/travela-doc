Image locationImage(BuildContext context, Location location) {
  if (location.photos.isEmpty) {
    return Image(image: GenericCard.getRandomImage(location));
  } else {
    return Image(
        image: CachedNetworkImageProvider(
            'https://maps.googleapis.com/maps/api/place/photo'
            '?maxwidth=${MediaQuery.of(context).size.height.toInt()}'
            '&maxHeight=${MediaQuery.of(context).size.width.toInt()}'
            '&photo_reference=${location.photos.first['photo_reference']}'
            '&key=${Keystore.googlePlaceAPIKey}'));
  }
}
